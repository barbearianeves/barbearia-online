function stripAccents(s){ return String(s || "") .normalize("NFD") .replace(/[\u0300-\u036f]/g, ""); } function normalizeTextInput(s){ return stripAccents(String(s || "")) .replace(/[^\w\s@.+\-]/g, " ") .replace(/_/g, " ") .replace(/\s+/g, " ") .trim(); } function normalizeNameInput(s){ return normalizeTextInput(s) .toLowerCase() .replace(/\b\w/g, c => c.toUpperCase()); } function normalizeEmailInput(s){ return stripAccents(String(s || "")) .toLowerCase() .replace(/\s+/g, "") .replace(/[^a-z0-9@._+\-]/g, "") .trim(); } function normalizePhone(v){ return String(v || "") .trim() .replace(/\D+/g, "") .replace(/^351(?=\d{9}$)/, ""); }
